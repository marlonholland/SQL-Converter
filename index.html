CREATE FUNCTION f_emp_info_2 (p_first_name VARCHAR(255), p_last_name VARCHAR(255)) RETURNS DECIMAL(10,2)
DETERMINISTIC
BEGIN
    DECLARE v_salary DECIMAL(10,2);
    DECLARE v_max_from_date DATE;
        
    SELECT  MAX(s.from_date)
    INTO  v_max_from_date
    FROM salaries s 
    LEFT JOIN employees e on s.emp_no = e.emp_no
    WHERE e.first_name = p_first_name AND e.last_name = p_last_name;
    
    SELECT s.salary 
    INTO v_salary
    FROM salaries s
    LEFT JOIN employees e on s.emp_no = e.emp_no
    WHERE s.from_date = v_max_from_date
    AND e.first_name = p_first_name 
    AND e.last_name = p_last_name;
    
    RETURN v_salary;
END$$
